pipeline {
    agent any

    tools {
        maven 'maven-3.9.14'
        jdk 'jdk-21'
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Clean & Compile') {
            steps {
                sh 'mvn clean compile'
            }
        }

        stage('Static Code Analysis (Fast)') {
            steps {
                // Только быстрая проверка, без генерации полных отчётов
                sh 'mvn spotbugs:check || exit 0'
                sh 'mvn checkstyle:check || exit 0'
                sh 'mvn pmd:check || exit 0'

                echo "Быстрый статический анализ выполнен"
            }
        }

        stage('Info') {
            steps {
                echo "📌 FEATURE ветка: только компиляция и анализ"
                echo "📌 JAR не создаётся для экономии времени"
            }
        }
    }

    post {
        success {
            echo "✅ FEATURE ветка: быстрая проверка пройдена"
        }
        failure {
            echo "❌ В feature ветке найдены проблемы. Исправьте перед merge в develop"
        }
    }
}
